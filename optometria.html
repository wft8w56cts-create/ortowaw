import sys import time import math import random import csv from threading import Thread from pythonosc.dispatcher import Dispatcher from pythonosc.osc_server import BlockingOSCServer import pygame # --- KONFIGURACJA SIECI --- IP_KOMPUTERA = "0.0.0.0" # Słuchaj wszystkich adresów w sieci lokalnej PORT = 5005 # --- GLOBALNE ZMIENNE (Z FACEID / ZIG SIM) --- eye_x, eye_y = 0.0, 0.0 raw_left_eye = [0.0, 0.0] # Monitorowanie ruchu głowy (pozycja nosa lub rotacja z ARKit) # ZIG SIM przesyła pozycję/rotację głowy w pakiecie ARKit head_moved = False base_head_x = None # Tłumaczenie mikroruchów oka na piksele ekranu SENSITIVITY_X = 1500 SENSITIVITY_Y = 1200 # Tolerancja ruchu głowy (im mniejsza liczba, tym trudniej/bardziej czule) HEAD_TOLERANCE = 0.025 # --- TELEMETRIA --- telemetry_log = [] session_start_time = time.time() # --- OBSŁUGA SYGNAŁU Z IPHONE (OSC) --- def faceid_handler(address, *args): global eye_x, eye_y, raw_left_eye, head_moved, base_head_x if "eye" in address or "face" in address: try: if len(args) >= 2: # 1. Pobranie pozycji wzroku raw_x = float(args[0]) raw_y = float(args[1]) raw_left_eye = [raw_x, raw_y] eye_x = raw_x * SENSITIVITY_X eye_y = raw_y * SENSITIVITY_Y # 2. Przykładowe sprawdzenie pozycji głowy, jeśli aplikacja wysyła pozycję twarzy (np. args[2] to pozycja X głowy) # Jeśli ZIG SIM podaje pozycję głowy w tym samym argumencie: if len(args) >= 3: current_head_x = float(args[2]) if base_head_x is None: base_head_x = current_head_x # Sprawdzamy czy głowa odchyliła się od punktu startowego if abs(current_head_x - base_head_x) > HEAD_TOLERANCE: head_moved = True else: head_moved = False except Exception: pass def start_osc_server(): dispatcher = Dispatcher() dispatcher.map("/*", faceid_handler) server = BlockingOSCServer((IP_KOMPUTERA, PORT), dispatcher) server.serve_forever() threading_server = Thread(target=start_osc_server, daemon=True) threading_server.start() # --- LOGIKA GRY (PYGAME) --- pygame.init() info = pygame.display.Info() SCREEN_W, SCREEN_H = info.current_w, info.current_h screen = pygame.display.set_mode((SCREEN_W, SCREEN_H), pygame.FULLSCREEN) pygame.display.set_caption("Trening Optometryczny - Ciągły Biofeedback") clock = pygame.time.Clock() gaze_x, gaze_y = SCREEN_W // 2, SCREEN_H // 2 # Obiekt: Rakieta rocket_x = random.randint(int(SCREEN_W * 0.2), int(SCREEN_W * 0.8)) rocket_y = random.randint(int(SCREEN_H * 0.2), int(SCREEN_H * 0.8)) rocket_radius = 30 lock_time = 0 score = 0 def save_to_csv(): if not telemetry_log: return filename = f"faceid_continuous_log_{int(time.time())}.csv" with open(filename, mode='w', newline='') as file: writer = csv.writer(file) writer.writerow(["Czas_ms", "Cel_X", "Cel_Y", "Wzrok_X", "Wzrok_Y", "Ruch_Glowy"]) writer.writerows(telemetry_log) print(f" Zapisano logi sesji do pliku: {filename}") running = True last_log_time = 0 print("Uruchomiono grę w trybie ciągłym. Oczekiwanie na FaceID...") while running: dt = clock.tick(60) current_time_ms = int((time.time() - session_start_time) * 1000) for event in pygame.event.get(): if event.type == pygame.QUIT or (event.type == pygame.KEYDOWN and event.key == pygame.K_ESCAPE): running = False # Klawisz R pozwala zresetować pozycję bazową głowy w trakcie gry if event.type == pygame.KEYDOWN and event.key == pygame.K_r: base_head_x = None # 1. Wygładzanie pozycji wzroku target_x = (SCREEN_W // 2) + eye_x target_y = (SCREEN_H // 2) - eye_y gaze_x += (target_x - gaze_x) * 0.15 gaze_y += (target_y - gaze_y) * 0.15 gaze_x = max(0, min(SCREEN_W, gaze_x)) gaze_y = max(0, min(SCREEN_H, gaze_y)) # 2. Nowe zasady naliczania punktów (Ciągła gra + warunek głowy) dist = math.hypot(gaze_x - rocket_x, gaze_y - rocket_y) is_looking_at_rocket = dist < 120 # PUNKTY ROSNĄ TYLKO GDY: patrzy na rakietę ORAZ głowa jest prosto if is_looking_at_rocket and not head_moved: lock_time += dt if lock_time >= 1500: # 1.5 sekundy poprawnej fiksacji score += 1 rocket_x = random.randint(int(SCREEN_W * 0.2), int(SCREEN_W * 0.8)) rocket_y = random.randint(int(SCREEN_H * 0.2), int(SCREEN_H * 0.8)) lock_time = 0 else: # Jeśli nie patrzy lub rusza głową, pasek postępu ładowania powoli spada lock_time = max(0, lock_time - dt) # 3. Zapis telemetrii (Z uwzględnieniem stanu głowy: 1 = ruch, 0 = stabilna) if current_time_ms - last_log_time >= 50: telemetry_log.append([ current_time_ms, rocket_x, rocket_y, int(gaze_x), int(gaze_y), 1 if head_moved else 0 ]) last_log_time = current_time_ms # 4. Rysowanie grafiki screen.fill((10, 10, 25)) # Kolor okręgu fiksacji: zielony jeśli wszystko ok, żółty jeśli zablokowany przez ruch głowy circle_color = (0, 255, 150) if not head_moved else (255, 200, 0) # Rysowanie wskaźnika ładowania fiksacji if lock_time > 0: pygame.draw.circle(screen, circle_color, (rocket_x, rocket_y), rocket_radius + 15, 3) # Rysowanie celu (Rakiety) pygame.draw.circle(screen, (255, 50, 100), (rocket_x, rocket_y), rocket_radius) # Rysowanie Latarki wzroku pygame.draw.circle(screen, (255, 255, 255), (int(gaze_x), int(gaze_y)), 100, 1) pygame.draw.circle(screen, (0, 255, 204), (int(gaze_x), int(gaze_y)), 8) # Teksty informacyjne i HUD font = pygame.font.SysFont("Segoe UI", 30) score_txt = font.render(f"Punkty: {score}", True, (255, 255, 255)) screen.blit(score_txt, (20, 20)) # Dynamiczny komunikat na dole ekranu w zależności od postawy dziecka if head_moved: status_txt = font.render("⚠️ UNIERUCHOM GŁOWĘ, aby zdobywać punkty!", True, (255, 200, 0)) else: status_txt = font.render("👍 Dobra postawa! Punktowanie aktywne.", True, (0, 255, 150)) help_txt = font.render("ESC - Wyjście i zapis CSV | R - Kalibracja głowy", True, (120, 120, 120)) screen.blit(status_txt, (20, SCREEN_H - 90)) screen.blit(help_txt, (20, SCREEN_H - 50)) pygame.display.flip() pygame.quit() save_to_csv() sys.exit()
